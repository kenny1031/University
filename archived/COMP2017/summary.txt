How to handle SIGRTMIN?

Response:
    SIGRTMIN is the first real-time signal in Unix-like systems (typically signal number 34 on Linux). Here's how to handle it:
    Basic Signal Handler Setup
    c#include <signal.h>
    #include <stdio.h>
    #include <unistd.h>

    void sigrtmin_handler(int sig, siginfo_t *info, void *context) {
        printf("Received SIGRTMIN (signal %d)\n", sig);
        printf("Sender PID: %d\n", info->si_pid);
        printf("Signal value: %d\n", info->si_value.sival_int);
    }

    int main() {
        struct sigaction sa;
        
        // Set up the signal handler
        sa.sa_sigaction = sigrtmin_handler;
        sigemptyset(&sa.sa_mask);
        sa.sa_flags = SA_SIGINFO;  // Use extended handler with siginfo_t
        
        if (sigaction(SIGRTMIN, &sa, NULL) == -1) {
            perror("sigaction");
            return 1;
        }
        
        printf("Waiting for SIGRTMIN... PID: %d\n", getpid());
        
        while (1) {
            pause();  // Wait for signals
        }
        
        return 0;
    }
    Key Points About SIGRTMIN
    Real-time signals (SIGRTMIN to SIGRTMAX) have special properties:

    They're queued (multiple instances can be pending)
    They carry additional data via sigqueue()
    They're delivered in priority order
    They support the SA_SIGINFO flag for extended information

    Sending SIGRTMIN with Data
    c#include <signal.h>
    #include <sys/types.h>

    // Send SIGRTMIN with custom data
    union sigval value;
    value.sival_int = 42;  // Custom integer data

    sigqueue(target_pid, SIGRTMIN, value);
    Python Example

        print(f"Received signal {signum}")

    Important Considerations

    Portability: SIGRTMIN value varies between systems
    Availability: Check if real-time signals are supported
    Handler safety: Use async-signal-safe functions only
    Masking: Consider blocking other signals during handler execution

    Real-time signals are particularly useful for IPC when you need reliable, ordered delivery with custom data payload.

    how to create a queue in C

    Response: Here are several ways to create queues in C, from simple arrays to more advanced implementations:
    1. Simple Array-Based Queue (Circular Buffer)
    c#include <stdio.h>
    #include <stdlib.h>
    #include <stdbool.h>

    #define MAX_SIZE 100

    typedef struct {
        int data[MAX_SIZE];
        int front;
        int rear;
        int size;
    } Queue;

    Queue* createQueue() {
        Queue* q = (Queue*)malloc(sizeof(Queue));
        q->front = 0;
        q->rear = -1;
        q->size = 0;
        return q;
    }

    bool isEmpty(Queue* q) {
        return q->size == 0;
    }

    bool isFull(Queue* q) {
        return q->size == MAX_SIZE;
    }

    bool enqueue(Queue* q, int value) {
        if (isFull(q)) {
            printf("Queue is full\n");
            return false;
        }
        
        q->rear = (q->rear + 1) % MAX_SIZE;
        q->data[q->rear] = value;
        q->size++;
        return true;
    }

    bool dequeue(Queue* q, int* value) {
        if (isEmpty(q)) {
            printf("Queue is empty\n");
            return false;
        }
        
        *value = q->data[q->front];
        q->front = (q->front + 1) % MAX_SIZE;
        q->size--;
        return true;
    }

    int peek(Queue* q) {
        if (isEmpty(q)) {
            printf("Queue is empty\n");
            return -1;
        }
        return q->data[q->front];
    }

    void destroyQueue(Queue* q) {
        free(q);
    }
    2. Dynamic Queue Using Linked List
    c#include <stdio.h>
    #include <stdlib.h>
    #include <stdbool.h>

    typedef struct Node {
        int data;
        struct Node* next;
    } Node;

    typedef struct {
        Node* front;
        Node* rear;
        int size;
    } Queue;

    Queue* createQueue() {
        Queue* q = (Queue*)malloc(sizeof(Queue));
        q->front = NULL;
        q->rear = NULL;
        q->size = 0;
        return q;
    }

    bool isEmpty(Queue* q) {
        return q->front == NULL;
    }

    bool enqueue(Queue* q, int value) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        if (!newNode) {
            printf("Memory allocation failed\n");
            return false;
        }
        
        newNode->data = value;
        newNode->next = NULL;
        
        if (isEmpty(q)) {
            q->front = q->rear = newNode;
        } else {
            q->rear->next = newNode;
            q->rear = newNode;
        }
        
        q->size++;
        return true;
    }

    bool dequeue(Queue* q, int* value) {
        if (isEmpty(q)) {
            printf("Queue is empty\n");
            return false;
        }
        
        Node* temp = q->front;
        *value = temp->data;
        q->front = q->front->next;
        
        if (q->front == NULL) {
            q->rear = NULL;  // Queue is now empty
        }
        
        free(temp);
        q->size--;
        return true;
    }

    int peek(Queue* q) {
        if (isEmpty(q)) {
            printf("Queue is empty\n");
            return -1;
        }
        return q->front->data;
    }

    void destroyQueue(Queue* q) {
        while (!isEmpty(q)) {
            int dummy;
            dequeue(q, &dummy);
        }
        free(q);
    }
    3. Generic Queue (Void Pointers)
    c#include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #include <stdbool.h>

    typedef struct Node {
        void* data;
        struct Node* next;
    } Node;

    typedef struct {
        Node* front;
        Node* rear;
        int size;
        size_t dataSize;
    } Queue;

    Queue* createQueue(size_t dataSize) {
        Queue* q = (Queue*)malloc(sizeof(Queue));
        q->front = NULL;
        q->rear = NULL;
        q->size = 0;
        q->dataSize = dataSize;
        return q;
    }

    bool enqueue(Queue* q, void* data) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        if (!newNode) return false;
        
        newNode->data = malloc(q->dataSize);
        if (!newNode->data) {
            free(newNode);
            return false;
        }
        
        memcpy(newNode->data, data, q->dataSize);
        newNode->next = NULL;
        
        if (q->rear == NULL) {
            q->front = q->rear = newNode;
        } else {
            q->rear->next = newNode;
            q->rear = newNode;
        }
        
        q->size++;
        return true;
    }

    bool dequeue(Queue* q, void* data) {
        if (q->front == NULL) return false;
        
        Node* temp = q->front;
        memcpy(data, temp->data, q->dataSize);
        
        q->front = q->front->next;
        if (q->front == NULL) {
            q->rear = NULL;
        }
        
        free(temp->data);
        free(temp);
        q->size--;
        return true;
    }
    4. Usage Example
    cint main() {
        // Array-based queue
        Queue* q1 = createQueue();
        
        enqueue(q1, 10);
        enqueue(q1, 20);
        enqueue(q1, 30);
        
        int value;
        while (dequeue(q1, &value)) {
            printf("Dequeued: %d\n", value);
        }
        
        destroyQueue(q1);
        
        // Generic queue with strings
        Queue* strQueue = createQueue(sizeof(char*));
        char* str1 = "Hello";
        char* str2 = "World";
        
        enqueue(strQueue, &str1);
        enqueue(strQueue, &str2);
        
        char* result;
        while (dequeue(strQueue, &result)) {
            printf("Dequeued string: %s\n", result);
        }
        
        destroyQueue(strQueue);
        
        return 0;
    }
    Key Design Decisions
    Array vs Linked List:

    Array: Fixed size, better cache locality, no dynamic allocation overhead
    Linked List: Dynamic size, more memory per element, better for unknown sizes

    Circular Buffer Benefits:

    Efficient use of array space
    Constant time operations
    No need to shift elements

    Generic Implementation:

    Works with any data type
    Uses void pointers and memcpy
    Requires caller to manage data types

    Choose the implementation based on your specific needs: fixed vs dynamic size, performance requirements, and whether you need generic functionality.
